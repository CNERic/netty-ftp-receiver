001package com.alexkasko.netty.ftp;
002
003import org.jboss.netty.buffer.ChannelBuffer;
004import org.jboss.netty.channel.Channel;
005import org.jboss.netty.channel.ChannelHandlerContext;
006import org.jboss.netty.handler.codec.frame.FrameDecoder;
007
008import java.nio.charset.Charset;
009
010/**
011 * {@link FrameDecoder} implementation, that accumulates input strings until {@code \r\n}
012 * and sends accumulated string upstream.
013 *
014 * @author alexkasko
015 * Date: 12/28/12
016 */
017public class CrlfStringDecoder extends FrameDecoder {
018    private static final byte CR = 13;
019    private static final byte LF = 10;
020
021    private final int maxRequestLengthBytes;
022    private final Charset encoding;
023
024    /**
025     * Constructor, uses {@code 256} max string length and {@code UTF-8} encoding
026     */
027    public CrlfStringDecoder() {
028        this(1<< 8, "UTF-8");
029    }
030
031    /**
032     * Constructor
033     *
034     * @param maxRequestLengthBytes max length of accumulated string in bytes
035     * @param encoding string encoding to use before sending it upstream
036     */
037    public CrlfStringDecoder(int maxRequestLengthBytes, String encoding) {
038        if(maxRequestLengthBytes <= 0) throw new IllegalArgumentException(
039                "Provided maxRequestLengthBytes: [" + maxRequestLengthBytes +"] must be positive");
040        this.maxRequestLengthBytes = maxRequestLengthBytes;
041        this.encoding = Charset.forName(encoding);
042    }
043
044    /**
045     * {@inheritDoc}
046     */
047    @Override
048    protected Object decode(ChannelHandlerContext ctx, Channel channel, ChannelBuffer cb) throws Exception {
049        byte[] data = new byte[maxRequestLengthBytes];
050        int lineLength = 0;
051        while (true) {
052            if (!cb.readable()) {
053                cb.resetReaderIndex();
054                return null;
055            }
056            byte nextByte = cb.readByte();
057            if (nextByte == CR) {
058                nextByte = cb.readByte();
059                if (nextByte == LF) {
060                    return new String(data, encoding);
061                }
062            } else if (nextByte == LF) {
063                return new String(data, encoding);
064            } else {
065                if (lineLength >= maxRequestLengthBytes) throw new IllegalArgumentException(
066                        "Request size threshold exceeded: [" + maxRequestLengthBytes + "]");
067                data[lineLength] = nextByte;
068                lineLength += 1;
069            }
070        }
071    }
072}
