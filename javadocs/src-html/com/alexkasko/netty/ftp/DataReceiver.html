001package com.alexkasko.netty.ftp;
002
003import java.io.IOException;
004import java.io.InputStream;
005
006/**
007 * Implementation should read all required data from provided FTP file-upload stream,
008 * stream will be closed immediately after {@link #receive(String, java.io.InputStream)} call
009 *
010 * @author alexkasko
011 * Date: 12/28/12
012 */
013public interface DataReceiver {
014    /**
015     * Implementation should read provided FTP file-upload data
016     *
017     * @param directory current directory set by client
018     * @param name name of uploaded file
019     * @param data uploaded file stream
020     * @throws IOException on IO error
021     */
022    void receive(String directory, String name, InputStream data) throws IOException;
023}
