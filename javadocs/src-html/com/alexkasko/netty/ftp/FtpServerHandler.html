015/**
016 * Netty handler, partial implementation of <a href="http://tools.ietf.org/html/rfc959">RFC 959 "File Transfer Protocol (FTP)"</a>
017 * for receiving FTP files. Both active and passive modes are supported.
018 *
019 * @author alexkasko
020 * Date: 12/27/12
021 */
022public class FtpServerHandler extends SimpleChannelUpstreamHandler {
023    private static final Logger logger = LoggerFactory.getLogger(FtpServerHandler.class);
024    private static final byte CR = 13;
025    private static final byte LF = 10;
026    private static final byte[] CRLF = new byte[]{CR, LF};
027    private static final Charset ASCII = Charset.forName("ASCII");
028
029    private final DataReceiver receiver;
030    private final byte[] passiveAddress;
031    private final byte[] passiveAdvertisedAddress;
032    private final int lowestPassivePort;
033    private final int highestPassivePort;
034    private final int passiveOpenAttempts;
035    private final int passiveSoTimeoutMillis;
036
037    // netty may be configured to use different worker threads with single handler
038    // even if handler is created for each pipeline
039    private AtomicReference<String> curDir = new AtomicReference<String>("/");
040    private AtomicReference<String> lastCommand = new AtomicReference<String>("");
041    private AtomicReference<Socket> activeSocket = new AtomicReference<Socket>();
042    private AtomicReference<ServerSocket> passiveSocket = new AtomicReference<ServerSocket>();
043
044    /**
045     * Constructor for FTP active mode
046     *
047     * @param receiver data receiver implementation
048     */
049    public FtpServerHandler(DataReceiver receiver) {
050        this(receiver, new byte[]{127, 0, 0, 1}, new byte[]{127, 0, 0, 1}, 2121, 4242, 10, 0);
051    }

053    /**
054     * Constructor for FTP passive mode
055     *
056     * @param receiver data receiver implementation
057     * @param passiveAddress passive IP address that will be used for server socket bind
058     * @param passiveAdvertisedAddress passive IP address that will be advertized to client
059     * @param lowestPassivePort lowest bound of passive ports range
060     * @param highestPassivePort highest bound of passive ports range
061     * @param passiveOpenAttempts number of ports to choose for passive socket open before reporting error
062     */
063    public FtpServerHandler(DataReceiver receiver, InetAddress passiveAddress, InetAddress passiveAdvertisedAddress,
064                            int lowestPassivePort, int highestPassivePort, int passiveOpenAttempts, int passiveSoTimeoutMillis) {
065        this(receiver, passiveAddress.getAddress(), passiveAdvertisedAddress.getAddress(), lowestPassivePort,
066                highestPassivePort, passiveOpenAttempts, passiveSoTimeoutMillis);
067    }

069    /**
070     * Constructor for FTP passive mode with separate advertized address
071     *
072     * @param receiver data receiver implementation
073     * @param passiveAddress passive IP address that will be used for server socket bind
074     * @param passiveAdvertisedAddress passive IP address that will be advertized to client
075     * @param lowestPassivePort lowest bound of passive ports range
076     * @param highestPassivePort highest bound of passive ports range
077     * @param passiveOpenAttempts number of ports to choose for passive socket open before reporting error
078     */
079    public FtpServerHandler(DataReceiver receiver, byte[] passiveAddress, byte[] passiveAdvertisedAddress,
080                            int lowestPassivePort, int highestPassivePort, int passiveOpenAttempts, int passiveSoTimeoutMillis) {
081        if(null == receiver) throw new IllegalArgumentException("Provided receiver is null");
082        this.receiver = receiver;
083        if(null == passiveAddress) throw new IllegalArgumentException("Provided passiveAddress is null");
084        this.passiveAddress = passiveAddress;
085        this.passiveAdvertisedAddress = passiveAdvertisedAddress;
086        if (lowestPassivePort <= 0 || lowestPassivePort >= 1 << 16) throw new IllegalArgumentException(
087                "Provided lowestPassivePort: [" + lowestPassivePort + "] ia out of valid range");
088        if (highestPassivePort <= 0 || highestPassivePort >= 1 << 16) throw new IllegalArgumentException(
089                        "Provided highestPassivePort: [" + highestPassivePort + "] ia out of valid range");
090        if (lowestPassivePort > highestPassivePort) throw new IllegalArgumentException(
091                "Provided lowestPassivePort: [" + lowestPassivePort + "] must be not greater than " +
092                        "highestPassivePort: [" + highestPassivePort + "]");
093        this.lowestPassivePort = lowestPassivePort;
094        this.highestPassivePort = highestPassivePort;
095        if(passiveOpenAttempts <= 0) throw new IllegalArgumentException(
096                "Provided passiveOpenAttempts: [" + passiveOpenAttempts + "] must be positive");
097        this.passiveOpenAttempts = passiveOpenAttempts;
098        if(passiveSoTimeoutMillis < 0) throw new IllegalArgumentException(
099                "Provided passiveSoTimeoutMillis: [" + passiveSoTimeoutMillis + "] must be positive");
100        this.passiveSoTimeoutMillis = passiveSoTimeoutMillis;
101    }

103    /**
104     * {@inheritDoc}
105     */
106    @Override
107    public void messageReceived(ChannelHandlerContext ctx, MessageEvent e) throws Exception {
108        String message = ((String) e.getMessage()).trim();
109        if (message.length() < 3) send("501 Syntax error", ctx, message, "");
110        String cmd = 3 == message.length() ? message.substring(0, 3) : message.substring(0, 4).trim();
111        String args = message.length() > cmd.length() ? message.substring(cmd.length() + 1) : "";
112        // dispatch
113        if ("USER".equals(cmd)) send("230 USER LOGGED IN", ctx, cmd, args);
114        else if ("CWD".equals(cmd)) { curDir.set(args); send("250 CWD command successful", ctx, cmd, args); }
115        else if ("PWD".equals(cmd)) send("257 \"" + curDir.get() + "\" is current directory", ctx, cmd, args);
116        else if ("MKD".equals(cmd)) send("521 \"" + args + "\" directory exists", ctx, cmd, args);
117        else if ("DELE".equals(cmd)) send("550 " + args + ": no such file or directory", ctx, cmd, args);
118        else if ("RMD".equals(cmd)) send("550 " + args + ": no such file or directory", ctx, cmd, args);
119        else if ("RNFR".equals(cmd)) send("350 File exists, ready for destination name", ctx, cmd, args);
120        else if ("RNTO".equals(cmd)) send("250 RNTO command successful", ctx, cmd, args);
121        else if ("SYST".equals(cmd)) send("215 UNIX Type: Java custom implementation", ctx, cmd, args);
122        else if ("NOOP".equals(cmd)) send("200 OK", ctx, cmd, args);
123        else if ("TYPE".equals(cmd)) type(ctx, args);
124        else if ("PORT".equals(cmd)) port(ctx, args);
125        else if ("PASV".equals(cmd)) pasv(ctx, args);
126        else if ("LIST".equals(cmd)) list(ctx, args);
127        else if ("STOR".equals(cmd)) stor(ctx, args);
128        else if ("QUIT".equals(cmd)) send("221 QUIT command successful", ctx, "QUIT", args);
129        else if ("ALLO".equals(cmd)) send("202 No storage allocation necessary", ctx, "ALLO", args);
130        else send("500 Command unrecognized", ctx, cmd, args);
131        lastCommand.set(cmd);
132    }

154     * @param ctx handler context

157    protected void type(ChannelHandlerContext ctx, String args) {
158        if ("I".equals(args)) send("200 Type set to IMAGE NONPRINT", ctx, "TYPE", args);
159        else if ("A".equals(args)) send("200 Type set to ASCII NONPRINT", ctx, "TYPE", args);
160        else send("504 Command not implemented for that parameter", ctx, "TYPE", args);

163    /**
164     * PORT command handler
165     *
166     * @param ctx handler context
167     * @param args command arguments
168     */
169    protected void port(ChannelHandlerContext ctx, String args) {
170        InetSocketAddress addr = parsePortArgs(args);
171        Socket as = activeSocket.get();
172        if (logger.isTraceEnabled()) logger.trace(String.valueOf(addr));
173        if (null == addr) send("501 Syntax error in parameters or arguments", ctx, "PORT", args);
174        else if (null != as) send("503 Bad sequence of commands", ctx, "PORT", args);
175        else {
176            try {
177                Socket created = new Socket(addr.getAddress(), addr.getPort());
178                boolean success = activeSocket.compareAndSet(null, created);
179                if(success) send("200 PORT command successful", ctx, "PORT", args);
180                else {
181                    logger.warn("Invalid concurrent handler usage detected");
182                    send("425 Server error", ctx, "PORT", args);
183                }
184            } catch (IOException e1) {
185                logger.warn("Exception thrown on opening active socket to address: [" + addr + "]", e1);
186                closeActiveSocket(null);
187                send("552 Requested file action aborted", ctx, "PORT", args);
188            }
189        }
190    }

192    /**
193     * PASV command handler
194     *
195     * @param ctx handler context
196     * @param args command arguments
197     * @throws InterruptedException
198     */
199    protected void pasv(ChannelHandlerContext ctx, String args) throws InterruptedException {
200        ServerSocket ps = passiveSocket.get();
201        if(null != ps) {
202            logger.warn("Invalid concurrent handler usage detected");
203            send("425 Server error", ctx, "PASV", args);
204            return;
205        }
206        for (int i = 0; i < passiveOpenAttempts; i++) {
207            int port = choosePassivePort(lowestPassivePort, highestPassivePort);
208            int part1 = (byte) (port >> 8) & 0xff;
209            int part2 = (byte) (port >> 0) & 0xff;
210            InetAddress addr = null;
211            try {
212                addr = InetAddress.getByAddress(passiveAddress);
213                ps = new ServerSocket(port, 0, addr);
214                ps.setSoTimeout(passiveSoTimeoutMillis);
215                send(String.format("227 Entering Passive Mode (%d,%d,%d,%d,%d,%d)",
216                        passiveAdvertisedAddress[0] & 0xff, passiveAdvertisedAddress[1] & 0xff,
217                        passiveAdvertisedAddress[2] & 0xff, passiveAdvertisedAddress[3] & 0xff,
218                        part1, part2), ctx, "PASV", args);
219                break;
220            } catch (IOException e1) {
221                logger.warn("Exception thrown on binding passive socket to address: [" + addr + "], port: [" + port + "], " +
222                        "attempt: [" + i + 1 + "] of: [" + passiveOpenAttempts + "]", e1);
223                closePassiveSocket(ps);
224                // ensure port change
225                Thread.sleep(1);
226            }
227        }
228        if(null != ps) {
229            boolean success = passiveSocket.compareAndSet(null, ps);
230            if(!success) {
231                logger.warn("Invalid concurrent handler usage detected");
232                send("425 Server error", ctx, "PASV", args);
233            }
234        }
235        else send("551 Requested action aborted", ctx, "PASV", args);
236    }

238    /**
239     * LIST command handler
240     *
241     * @param ctx handler context
242     * @param args command arguments
243     */
244    protected void list(ChannelHandlerContext ctx, String args) {
245        if ("PORT".equals(lastCommand.get())) {
246            Socket as = activeSocket.get();
247            if (null != as) {
248                send("150 Opening binary mode data connection for LIST " + args, ctx, "LIST", args);
249                try {
250                    as.getOutputStream().write(CRLF);
251                    send("226 Transfer complete for LIST", ctx, "", args);
252                } catch (IOException e1) {
253                    logger.warn("Exception thrown on writing through active socket: [" + as + "]", e1);
254                    send("552 Requested file action aborted", ctx, "LIST", args);
255                } finally {
256                    closeActiveSocket(as);
257                }
258            } else send("503 Bad sequence of commands", ctx, "LIST", args);
259        } else if ("PASV".equals(lastCommand.get())) {
260            ServerSocket ps = this.passiveSocket.get();
261            if (null != ps) {
262                send("150 Opening binary mode data connection for LIST on port: " + ps, ctx, "LIST", args);
263                Socket clientSocket = null;
264                try {
265                    clientSocket = ps.accept();
266                    clientSocket.getOutputStream().write(CRLF);
267                    clientSocket.getOutputStream().close();
268                    send("226 Transfer complete for LIST", ctx, "", args);
269                } catch (IOException e1) {
270                    logger.warn("Exception thrown on writing through passive socket: [" + ps + "]," +
271                            "accepted client socket: [" + clientSocket + "]", e1);
272                    send("552 Requested file action aborted", ctx, "LIST", args);
273                } finally {
274                    closePassiveSocket(ps);
275                }
276            } else send("503 Bad sequence of commands", ctx, "LIST", args);
277        } else send("503 Bad sequence of commands", ctx, "LIST", args);
278    }

280    /**
281     * STOP command handler
282     *
283     * @param ctx handler context
284     * @param args command arguments
285     */
286    protected void stor(ChannelHandlerContext ctx, String args) {
287        if ("PORT".equals(lastCommand.get())) {
288            Socket as = activeSocket.get();
289            if (null != as) {
290                send("150 Opening binary mode data connection for " + args, ctx, "", args);
291                try {
292                    receiver.receive(curDir.get(), args, as.getInputStream());
293                    send("226 Transfer complete for STOR " + args, ctx, "", args);
294                } catch (IOException e1) {
295                    logger.warn("Exception thrown on reading through active socket: [" + as + "]", e1);
296                    send("552 Requested file action aborted", ctx, "STOR", args);
297                } finally {
298                    closeActiveSocket(as);
299                }
300            } else send("503 Bad sequence of commands", ctx, "STOR", args);
301        } else if ("PASV".equals(lastCommand.get())) {
302            ServerSocket ps = this.passiveSocket.get();
303            if (null != ps) {
304                send("150 Opening binary mode data connection for " + args, ctx, "STOR", args);
305                Socket clientSocket = null;
306                try {
307                    clientSocket = ps.accept();
308                    receiver.receive(curDir.get(), args, clientSocket.getInputStream());
309                    send("226 Transfer complete for STOR " + args, ctx, "", args);
310                } catch (IOException e1) {
311                    logger.warn("Exception thrown on reading through passive socket: [" + ps + "], " +
312                            "accepted client socket: [" + clientSocket + "]", e1);
313                    send("552 Requested file action aborted", ctx, "STOR", args);
314                } finally {
315                    closePassiveSocket(ps);
316                }
317            } else send("503 Bad sequence of commands", ctx, "STOR", args);
318        } else send("503 Bad sequence of commands", ctx, "STOR", args);
319    }

321    private static void send(String response, ChannelHandlerContext ctx, String command, String args) {
322        if(logger.isDebugEnabled()) {
323            if(command.length() > 0) logger.debug("-> " + command + " " + args);
324            logger.debug("<- " + response);
325        }
326        String line = response + "\r\n";
327        byte[] data = line.getBytes(ASCII);
328        ctx.getChannel().write(wrappedBuffer(data));
329    }

331    private static InetSocketAddress parsePortArgs(String portArgs) {
332        String[] strParts = portArgs.split(",");
333        if (strParts.length != 6) return null;
334        byte[] address = new byte[4];
335        int[] parts = new int[6];
336        for (int i = 0; i < 6; i++) {
337            try { parts[i] = Integer.parseInt(strParts[i]); }
338            catch (NumberFormatException e) { return null; }
339            if (parts[i] < 0 || parts[i] > 255) return null;
340        }
341        for (int i = 0; i < 4; i++) address[i] = (byte) parts[i];
342        int port = parts[4] << 8 | parts[5];
343        InetAddress inetAddress;
344        try { inetAddress = InetAddress.getByAddress(address); }
345        catch (UnknownHostException e) { return null; }
346        return new InetSocketAddress(inetAddress, port);
347    }
348
349    private static int choosePassivePort(int low, int high) {
350        int length = high - low;
351        int offset = (int) (currentTimeMillis() % length);
352        return low + offset;
353    }
354
355    private void closeActiveSocket(Socket socket) {
356        Socket as = null != socket ? socket : activeSocket.get();
357        if(null == as) return;
358        try {
359            as.close();
360        } catch (Exception e) {
361            logger.warn("Exception thrown on closing active socket", e);
362        } finally {
363            boolean success = activeSocket.compareAndSet(as, null);
364            if(!success) logger.warn("Invalid concurrent handler usage detected");
365        }
366    }
367
368    private void closePassiveSocket(ServerSocket socket) {
369        ServerSocket ps = null != socket ? socket: passiveSocket.get();
370        if(null == ps) return;
371        try {
372            ps.close();
373        } catch (Exception e) {
374            logger.warn("Exception thrown on closing server socket", e);
375        } finally {
376            boolean success = passiveSocket.compareAndSet(ps, null);
377            if(!success) logger.warn("Invalid concurrent handler usage detected");
378        }
379    }
