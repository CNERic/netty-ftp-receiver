014/**
015 * Netty handler, partial implementation of <a href="http://tools.ietf.org/html/rfc959">RFC 959 "File Transfer Protocol (FTP)"</a>
016 * for receiving FTP files. Both active and passive modes are supported.
017 *
018 * @author alexkasko
019 * Date: 12/27/12
020 */
021public class FtpServerHandler extends SimpleChannelUpstreamHandler {
022    private static final Logger logger = LoggerFactory.getLogger(FtpServerHandler.class);
023    private static final byte CR = 13;
024    private static final byte LF = 10;
025    private static final byte[] CRLF = new byte[]{CR, LF};
026    private static final Charset ASCII = Charset.forName("ASCII");
027
028    private final DataReceiver receiver;
029    private final byte[] passiveAddress;
030    private final int lowestPassivePort;
031    private final int highestPassivePort;
032    private final int passiveOpenAttempts;
033
034    private String curDir = "/";
035    private String lastCommand = "";
036    private Socket activeSocket = null;
037    private ServerSocket passiveSocket = null;
038
039    /**
040     * Constructor for FTP active mode
041     *
042     * @param receiver data receiver implementation
043     */
044    public FtpServerHandler(DataReceiver receiver) {
045        this(receiver, new byte[]{127, 0, 0, 1}, 2121, 4242, 10);
046    }
047
048    /**
049     * Constructor for FTP passive mode
050     *
051     * @param receiver data receiver implementation
052     * @param passiveAddress passive IP address
053     * @param lowestPassivePort lowest bound of passive ports range
054     * @param highestPassivePort highest bound of passive ports range
055     * @param passiveOpenAttempts number of ports to choose for passive socket open before reporting error
056     */
057    public FtpServerHandler(DataReceiver receiver, InetAddress passiveAddress, int lowestPassivePort, int highestPassivePort, int passiveOpenAttempts) {
058        this(receiver, passiveAddress.getAddress(), lowestPassivePort, highestPassivePort, passiveOpenAttempts);
059    }
060
061    /**
062     * Constructor for FTP passive mode
063     *
064     * @param receiver data receiver implementation
065     * @param passiveAddress passive IP address
066     * @param lowestPassivePort lowest bound of passive ports range
067     * @param highestPassivePort highest bound of passive ports range
068     * @param passiveOpenAttempts number of ports to choose for passive socket open before reporting error
069     */
070    public FtpServerHandler(DataReceiver receiver, byte[] passiveAddress, int lowestPassivePort, int highestPassivePort, int passiveOpenAttempts) {
071        if(null == receiver) throw new IllegalArgumentException("Provided receiver is null");
072        this.receiver = receiver;
073        if(null == passiveAddress) throw new IllegalArgumentException("Provided passiveAddress is null");
074        this.passiveAddress = passiveAddress;
075        if (lowestPassivePort <= 0 || lowestPassivePort >= 1 << 16) throw new IllegalArgumentException(
076                "Provided lowestPassivePort: [" + lowestPassivePort + "] ia out of valid range");
077        if (highestPassivePort <= 0 || highestPassivePort >= 1 << 16) throw new IllegalArgumentException(
078                        "Provided highestPassivePort: [" + highestPassivePort + "] ia out of valid range");
079        if (lowestPassivePort > highestPassivePort) throw new IllegalArgumentException(
080                "Provided lowestPassivePort: [" + lowestPassivePort + "] must be not greater than " +
081                        "highestPassivePort: [" + highestPassivePort + "]");
082        this.lowestPassivePort = lowestPassivePort;
083        this.highestPassivePort = highestPassivePort;
084        if(passiveOpenAttempts <= 0) throw new IllegalArgumentException(
085                "Provided passiveOpenAttempts: [" + passiveOpenAttempts + "] must be positive");
086        this.passiveOpenAttempts = passiveOpenAttempts;
087    }
088
089    /**
090     * {@inheritDoc}
091     */
092    @Override
093    public void messageReceived(ChannelHandlerContext ctx, MessageEvent e) throws Exception {
094        String message = ((String) e.getMessage()).trim();
095        if (message.length() < 3) send("501 Syntax error", ctx, message, "");
096        String cmd = 3 == message.length() ? message.substring(0, 3) : message.substring(0, 4).trim();
097        String args = message.length() > cmd.length() ? message.substring(cmd.length() + 1) : "";
098        // dispatch
099        if ("USER".equals(cmd)) send("230 USER LOGGED IN", ctx, cmd, args);
100        else if ("CWD".equals(cmd)) { curDir = args; send("250 CWD command successful", ctx, cmd, args); }
101        else if ("PWD".equals(cmd)) send("257 \"" + curDir + "\" is current directory", ctx, cmd, args);
102        else if ("MKD".equals(cmd)) send("521 \"" + args + "\" directory exists", ctx, cmd, args);
103        else if ("DELE".equals(cmd)) send("550 " + args + ": no such file or directory", ctx, cmd, args);
104        else if ("RMD".equals(cmd)) send("550 " + args + ": no such file or directory", ctx, cmd, args);
105        else if ("RNFR".equals(cmd)) send("350 File exists, ready for destination name", ctx, cmd, args);
106        else if ("RNTO".equals(cmd)) send("250 RNTO command successful", ctx, cmd, args);
107        else if ("SYST".equals(cmd)) send("215 UNIX Type: Java custom implementation", ctx, cmd, args);
108        else if ("NOOP".equals(cmd)) send("200 OK", ctx, cmd, args);
109        else if ("TYPE".equals(cmd)) type(ctx, args);
110        else if ("PORT".equals(cmd)) port(ctx, args);
111        else if ("PASV".equals(cmd)) pasv(ctx, args);
112        else if ("LIST".equals(cmd)) list(ctx, args);
113        else if ("STOR".equals(cmd)) stor(ctx, args);
114        else if ("QUIT".equals(cmd)) send("221 QUIT command successful", ctx, "QUIT", args);
115        else send("500 Command unrecognized", ctx, cmd, args);
116        lastCommand = cmd;
117    }

136    /**
137     * TYPE command handler
138     *
139     * @param ctx handler context
140     * @param args command arguments
141     */
142    protected void type(ChannelHandlerContext ctx, String args) {
143        if ("I".equals(args)) send("200 Type set to IMAGE NONPRINT", ctx, "TYPE", args);
144        else if ("A".equals(args)) send("200 Type set to ASCII NONPRINT", ctx, "TYPE", args);
145        else send("504 Command not implemented for that parameter", ctx, "TYPE", args);
146    }
147
148    /**
149     * PORT command handler
150     *
151     * @param ctx handler context
152     * @param args command arguments
153     */
154    protected void port(ChannelHandlerContext ctx, String args) {
155        InetSocketAddress addr = parsePortArgs(args);
156        if (logger.isTraceEnabled()) logger.trace(String.valueOf(addr));
157        if (null == addr) send("501 Syntax error in parameters or arguments", ctx, "PORT", args);
158        else if (null != this.activeSocket && !activeSocket.isClosed()) send("503 Bad sequence of commands", ctx, "PORT", args);
159        else {
160            try {
161                this.activeSocket = new Socket(addr.getAddress(), addr.getPort());
162                send("200 PORT command successful", ctx, "PORT", args);
163            } catch (IOException e1) {
164                logger.warn("Exception thrown on opening active socket to address: [" + addr + "]", e1);
165                closeActiveSocket();
166                send("552 Requested file action aborted", ctx, "PORT", args);
167            }
168        }
169    }
170
171    /**
172     * PASV command handler
173     *
174     * @param ctx handler context
175     * @param args command arguments
176     * @throws InterruptedException
177     */
178    protected void pasv(ChannelHandlerContext ctx, String args) throws InterruptedException {
179        for (int i = 0; i < passiveOpenAttempts; i++) {
180            int port = choosePassivePort(lowestPassivePort, highestPassivePort);
181            int part1 = (byte) (port >> 8) & 0xff;
182            int part2 = (byte) (port >> 0) & 0xff;
183            InetAddress addr = null;
184            try {
185                addr = InetAddress.getByAddress(passiveAddress);
186                this.passiveSocket = new ServerSocket(port, 50, addr);
187                send(String.format("227 Entering Passive Mode (%d,%d,%d,%d,%d,%d)",
188                        passiveAddress[0], passiveAddress[1], passiveAddress[2], passiveAddress[3], part1, part2), ctx, "PASV", args);
189                break;
190            } catch (IOException e1) {
191                logger.warn("Exception thrown on binding passive socket to address: [" + addr + "], port: [" + port + "], " +
192                        "attempt: [" + i + 1 + "] of: [" + passiveOpenAttempts + "]", e1);
193                closePassiveSocket();
194                // ensure port change
195                Thread.sleep(1);
196            }
197        }
198        if(null == this.passiveSocket) send("551 Requested action aborted", ctx, "PASV", args);
199    }

201    /**
202     * LIST command handler
203     *
204     * @param ctx handler context
205     * @param args command arguments
206     */
207    protected void list(ChannelHandlerContext ctx, String args) {
208        if ("PORT".equals(lastCommand)) {
209            if (null == this.activeSocket) send("503 Bad sequence of commands", ctx, "LIST", args);
210            send("150 Opening binary mode data connection for LIST " + args, ctx, "LIST", args);
211            try {
212                activeSocket.getOutputStream().write(CRLF);
213                send("226 Transfer complete for LIST", ctx, "", args);
214            } catch (IOException e1) {
215                logger.warn("Exception thrown on writing through active socket: [" + activeSocket + "]", e1);
216                send("552 Requested file action aborted", ctx, "LIST", args);
217            } finally {
218                closeActiveSocket();
219            }
220        } else if ("PASV".equals(lastCommand)) {
221            if (null == this.passiveSocket) send("503 Bad sequence of commands", ctx, "LIST", args);
222            send("150 Opening binary mode data connection for LIST on port: " + passiveSocket.getLocalPort(), ctx, "LIST", args);
223            Socket clientSocket = null;
224            try {
225                clientSocket = passiveSocket.accept();
226                clientSocket.getOutputStream().write(CRLF);
227                clientSocket.getOutputStream().close();
228                send("226 Transfer complete for LIST", ctx, "", args);
229            } catch (IOException e1) {
230                logger.warn("Exception thrown on writing through passive socket: [" + passiveSocket + "]," +
231                        "accepted client socket: [" + clientSocket + "]", e1);
232                send("552 Requested file action aborted", ctx, "LIST", args);
233            } finally {
234                closePassiveSocket();
235            }
236        } else send("503 Bad sequence of commands", ctx, "LIST", args);
237    }

239    /**
240     * STOP command handler
241     *
242     * @param ctx handler context
243     * @param args command arguments
244     */
245    protected void stor(ChannelHandlerContext ctx, String args) {
246        if ("PORT".equals(lastCommand)) {
247            if (null == this.activeSocket) send("503 Bad sequence of commands", ctx, "STOR", args);
248            send("150 Opening binary mode data connection for " + args, ctx, "", args);
249            try {
250                receiver.receive(curDir, args, activeSocket.getInputStream());
251                send("226 Transfer complete for STOR " + args, ctx, "", args);
252            } catch (IOException e1) {
253                logger.warn("Exception thrown on reading through active socket: [" + activeSocket + "]", e1);
254                send("552 Requested file action aborted", ctx, "STOR", args);
255            } finally {
256                closeActiveSocket();
257            }
258        } else if ("PASV".equals(lastCommand)) {
259            if (null == this.passiveSocket) send("503 Bad sequence of commands", ctx, "STOR", args);
260            send("150 Opening binary mode data connection for " + args, ctx, "STOR", args);
261            Socket clientSocket = null;
262            try {
263                clientSocket = passiveSocket.accept();
264                receiver.receive(curDir, args, clientSocket.getInputStream());
265                send("226 Transfer complete for STOR " + args, ctx, "", args);
266            } catch (IOException e1) {
267                logger.warn("Exception thrown on reading through passive socket: [" + passiveSocket + "], " +
268                        "accepted client socket: [" + clientSocket + "]", e1);
269                send("552 Requested file action aborted", ctx, "STOR", args);
270            } finally {
271                closePassiveSocket();
272            }
273        } else send("503 Bad sequence of commands", ctx, "STOR", args);
274    }

276    private static void send(String response, ChannelHandlerContext ctx, String command, String args) {
277        if(logger.isDebugEnabled()) {
278            if(command.length() > 0) logger.debug("-> " + command + " " + args);
279            logger.debug("<- " + response);
280        }
281        String line = response + "\r\n";
282        byte[] data = line.getBytes(ASCII);
283        ctx.getChannel().write(wrappedBuffer(data));
284    }

286    private static InetSocketAddress parsePortArgs(String portArgs) {
287        String[] strParts = portArgs.split(",");
288        if (strParts.length != 6) return null;
289        byte[] address = new byte[4];
290        int[] parts = new int[6];
291        for (int i = 0; i < 6; i++) {
292            try { parts[i] = Integer.parseInt(strParts[i]); }
293            catch (NumberFormatException e) { return null; }
294            if (parts[i] < 0 || parts[i] > 255) return null;
295        }
296        for (int i = 0; i < 4; i++) address[i] = (byte) parts[i];
297        int port = parts[4] << 8 | parts[5];
298        InetAddress inetAddress;
299        try { inetAddress = InetAddress.getByAddress(address); }
300        catch (UnknownHostException e) { return null; }
301        return new InetSocketAddress(inetAddress, port);
302    }
303
304    private static int choosePassivePort(int low, int high) {
305        int length = high - low;
306        int offset = (int) (currentTimeMillis() % length);
307        return low + offset;
308    }
309
310    private void closeActiveSocket() {
311        if(null == activeSocket) return;
312        try {
313            activeSocket.close();
314        } catch (Exception e) {
315            logger.warn("Exception thrown on closing active socket", e);
316        } finally {
317            activeSocket = null;
318        }
319    }
320
321    private void closePassiveSocket() {
322        if(null == passiveSocket) return;
323        try {
324            passiveSocket.close();
325        } catch (Exception e) {
326            logger.warn("Exception thrown on closing server socket", e);
327        } finally {
328            passiveSocket = null;
329        }
330    }
