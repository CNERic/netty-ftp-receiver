014    /**
015     * Netty handler, partial implementation of <a href="http://tools.ietf.org/html/rfc959">RFC 959 "File Transfer Protocol (FTP)"</a>
016     * for receiving FTP files. Both active and passive modes are supported.
017     *
018     * @author alexkasko
019     * Date: 12/27/12
020     */
021    public class FtpServerHandler extends SimpleChannelUpstreamHandler {
022        private static final Logger logger = LoggerFactory.getLogger(FtpServerHandler.class);
023        private static final byte CR = 13;
024        private static final byte LF = 10;
025        private static final byte[] CRLF = new byte[]{CR, LF};
026        private static final Charset ASCII = Charset.forName("ASCII");
027    
028        private final DataReceiver receiver;
029        private final byte[] passiveAddress;
030        private final int lowestPassivePort;
031        private final int highestPassivePort;
032        private final int passiveOpenAttempts;
033    
034        private String curDir = "/";
035        private String lastCommand = "";
036        private Socket activeSocket = null;
037        private ServerSocket passiveSocket = null;
038    
039        /**
040         * Constructor for FTP active mode
041         *
042         * @param receiver data receiver implementation
043         */
044        public FtpServerHandler(DataReceiver receiver) {
045            this(receiver, new byte[]{127, 0, 0, 1}, 2121, 4242, 10);
046        }
047    
048        /**
049         * Constructor for FTP passive mode
050         *
051         * @param receiver data receiver implementation
052         * @param passiveAddress passive IP address
053         * @param lowestPassivePort lowest bound of passive ports range
054         * @param highestPassivePort highest bound of passive ports range
055         * @param passiveOpenAttempts number of ports to choose for passive socket open before reporting error
056         */
057        public FtpServerHandler(DataReceiver receiver, InetAddress passiveAddress, int lowestPassivePort, int highestPassivePort, int passiveOpenAttempts) {
058            this(receiver, passiveAddress.getAddress(), lowestPassivePort, highestPassivePort, passiveOpenAttempts);
059        }
060    
061        /**
062         * Constructor for FTP passive mode
063         *
064         * @param receiver data receiver implementation
065         * @param passiveAddress passive IP address
066         * @param lowestPassivePort lowest bound of passive ports range
067         * @param highestPassivePort highest bound of passive ports range
068         * @param passiveOpenAttempts number of ports to choose for passive socket open before reporting error
069         */
070        public FtpServerHandler(DataReceiver receiver, byte[] passiveAddress, int lowestPassivePort, int highestPassivePort, int passiveOpenAttempts) {
071            if(null == receiver) throw new IllegalArgumentException("Provided receiver is null");
072            this.receiver = receiver;
073            if(null == passiveAddress) throw new IllegalArgumentException("Provided passiveAddress is null");
074            this.passiveAddress = passiveAddress;
075            if (lowestPassivePort <= 0 || lowestPassivePort >= 1 << 16) throw new IllegalArgumentException(
076                    "Provided lowestPassivePort: [" + lowestPassivePort + "] ia out of valid range");
077            if (highestPassivePort <= 0 || highestPassivePort >= 1 << 16) throw new IllegalArgumentException(
078                            "Provided highestPassivePort: [" + highestPassivePort + "] ia out of valid range");
079            if (lowestPassivePort > highestPassivePort) throw new IllegalArgumentException(
080                    "Provided lowestPassivePort: [" + lowestPassivePort + "] must be not greater than " +
081                            "highestPassivePort: [" + highestPassivePort + "]");
082            this.lowestPassivePort = lowestPassivePort;
083            this.highestPassivePort = highestPassivePort;
084            if(passiveOpenAttempts <= 0) throw new IllegalArgumentException(
085                    "Provided passiveOpenAttempts: [" + passiveOpenAttempts + "] must be positive");
086            this.passiveOpenAttempts = passiveOpenAttempts;
087        }
088    
089        /**
090         * {@inheritDoc}
091         */
092        @Override
093        public void messageReceived(ChannelHandlerContext ctx, MessageEvent e) throws Exception {
094            String message = ((String) e.getMessage()).trim();
095            if (message.length() < 3) send("501 Syntax error", ctx, message, "");
096            String cmd = 3 == message.length() ? message.substring(0, 3) : message.substring(0, 4).trim();
097            String args = message.length() > cmd.length() ? message.substring(cmd.length() + 1) : "";
098            // dispatch
099            if ("USER".equals(cmd)) send("230 USER LOGGED IN", ctx, cmd, args);
100            else if ("CWD".equals(cmd)) { curDir = args; send("250 CWD command successful", ctx, cmd, args); }
101            else if ("PWD".equals(cmd)) send("257 \"" + curDir + "\" is current directory", ctx, cmd, args);
102            else if ("MKD".equals(cmd)) send("521 \"" + args + "\" directory exists", ctx, cmd, args);
103            else if ("DELE".equals(cmd)) send("550 " + args + ": no such file or directory", ctx, cmd, args);
104            else if ("RMD".equals(cmd)) send("550 " + args + ": no such file or directory", ctx, cmd, args);
105            else if ("RNFR".equals(cmd)) send("350 File exists, ready for destination name", ctx, cmd, args);
106            else if ("RNTO".equals(cmd)) send("250 RNTO command successful", ctx, cmd, args);
107            else if ("SYST".equals(cmd)) send("215 UNIX Type: Java custom implementation", ctx, cmd, args);
108            else if ("NOOP".equals(cmd)) send("200 OK", ctx, cmd, args);
109            else if ("TYPE".equals(cmd)) type(ctx, args);
110            else if ("PORT".equals(cmd)) port(ctx, args);
111            else if ("PASV".equals(cmd)) pasv(ctx, args);
112            else if ("LIST".equals(cmd)) list(ctx, args);
113            else if ("STOR".equals(cmd)) stor(ctx, args);
114            else send("500 Command unrecognized", ctx, cmd, args);
115            lastCommand = cmd;
116        }

121        @Override
122        public void exceptionCaught(ChannelHandlerContext ctx, ExceptionEvent e) throws Exception {
123            e.getCause().printStackTrace();
124            send("500 Unspecified error", ctx, e.getCause().getMessage(), "");
125        }

132            send("220 Service ready", ctx, "[connected]", "");

135        /**
136         * TYPE command handler
137         *
138         * @param ctx handler context
139         * @param args command arguments
140         */
141        protected void type(ChannelHandlerContext ctx, String args) {
142            if ("I".equals(args)) send("200 Type set to IMAGE NONPRINT", ctx, "TYPE", args);
143            else if ("A".equals(args)) send("200 Type set to ASCII NONPRINT", ctx, "TYPE", args);
144            else send("504 Command not implemented for that parameter", ctx, "TYPE", args);
145        }
146    
147        /**
148         * PORT command handler
149         *
150         * @param ctx handler context
151         * @param args command arguments
152         */
153        protected void port(ChannelHandlerContext ctx, String args) {
154            InetSocketAddress addr = parsePortArgs(args);
155            if (null == addr) send("501 Syntax error in parameters or arguments", ctx, "PORT", args);
156            else if (null != this.activeSocket && !activeSocket.isClosed()) send("503 Bad sequence of commands", ctx, "PORT", args);
157            else {
158                try {
159                    this.activeSocket = new Socket(addr.getAddress(), addr.getPort());
160                    send("200 PORT command successful", ctx, "PORT", args);
161                } catch (IOException e1) {
162                    send("552 Requested file action aborted", ctx, "PORT", args);
163                }
164            }
165        }
166    
167        /**
168         * PASV command handler
169         *
170         * @param ctx handler context
171         * @param args command arguments
172         * @throws InterruptedException
173         */
174        protected void pasv(ChannelHandlerContext ctx, String args) throws InterruptedException {
175            for(int i=0; i< passiveOpenAttempts; i++) {
176                int port = choosePassivePort(lowestPassivePort, highestPassivePort);
177                int part1 = (byte) (port >> 8) & 0xff;
178                int part2 = (byte) (port >> 0) & 0xff;
179                try {
180                    InetAddress addr = InetAddress.getByAddress(passiveAddress);
181                    this.passiveSocket = new ServerSocket(port, 50, addr);
182                    send(String.format("227 Entering Passive Mode (%d,%d,%d,%d,%d,%d)",
183                            passiveAddress[0], passiveAddress[1], passiveAddress[2], passiveAddress[3], part1, part2), ctx, "PASV", args);
184                    break;
185                } catch (IOException e1) {
186                    logger.warn("Fail binding on port: " + port);
187                    // ensure port change
188                    Thread.sleep(1);
189                }
190            }
191            if(null == this.passiveSocket) send("551 Requested action aborted", ctx, "PASV", args);
192        }

194        /**
195         * LIST command handler
196         *
197         * @param ctx handler context
198         * @param args command arguments
199         */
200        protected void list(ChannelHandlerContext ctx, String args) {
201            if ("PORT".equals(lastCommand)) {
202                send("150 Opening binary mode data connection for LIST " + args, ctx, "LIST", args);
203                try {
204                    activeSocket.getOutputStream().write(CRLF);
205                    send("226 Transfer complete for LIST", ctx, "", args);
206                } catch (IOException e1) {
207                    send("552 Requested file action aborted", ctx, "LIST", args);
208                } finally {
209                    closeQuetly(activeSocket);
210                    activeSocket = null;
211                }
212            } else if ("PASV".equals(lastCommand)) {
213                send("150 Opening binary mode data connection for LIST on port: " + passiveSocket.getLocalPort(), ctx, "LIST", args);
214                try {
215                    Socket clientSocket = passiveSocket.accept();
216                    clientSocket.getOutputStream().write(CRLF);
217                    clientSocket.getOutputStream().close();
218                    send("226 Transfer complete for LIST", ctx, "", args);
219                } catch (IOException e1) {
220                    send("552 Requested file action aborted", ctx, "LIST", args);
221                } finally {
222                    closeQuetly(passiveSocket);
223                    passiveSocket = null;
224                }
225            } else send("503 Bad sequence of commands", ctx, "LIST", args);
226        }
227    
228        /**
229         * STOP command handler
230         *
231         * @param ctx handler context
232         * @param args command arguments
233         */
234        protected void stor(ChannelHandlerContext ctx, String args) {
235            if ("PORT".equals(lastCommand)) {
236                send("150 Opening binary mode data connection for " + args, ctx, "", args);
237                try {
238                    receiver.receive(args, activeSocket.getInputStream());
239                    send("226 Transfer complete for STOR " + args, ctx, "", args);
240                } catch (IOException e1) {
241                    send("552 Requested file action aborted", ctx, "STOR", args);
242                } finally {
243                    closeQuetly(activeSocket);
244                    activeSocket = null;
245                }
246            } else if ("PASV".equals(lastCommand)) {
247                send("150 Opening binary mode data connection for " + args, ctx, "STOR", args);
248                try {
249                    Socket clientSocket = passiveSocket.accept();
250                    receiver.receive(args, clientSocket.getInputStream());
251                    send("226 Transfer complete for STOR " + args, ctx, "", args);
252                } catch (IOException e1) {
253                    send("552 Requested file action aborted", ctx, "STOR", args);
254                } finally {
255                    closeQuetly(passiveSocket);
256                    passiveSocket = null;
257                }
258            } else send("503 Bad sequence of commands", ctx, "STOR", args);
259        }

261        private static void send(String response, ChannelHandlerContext ctx, String command, String args) {
262            if(command.length() > 0) logger.debug("-> " + command + " " + args);
263            logger.debug("<- " + response);
264            String line = response + "\r\n";
265            byte[] data = line.getBytes(ASCII);
266            ctx.getChannel().write(wrappedBuffer(data));
267        }

269        private static InetSocketAddress parsePortArgs(String portArgs) {
270            String[] strParts = portArgs.split(",");
271            if (strParts.length != 6) return null;
272            byte[] address = new byte[4];
273            int[] parts = new int[6];
274            for (int i = 0; i < 6; i++) {
275                try { parts[i] = Integer.parseInt(strParts[i]); }
276                catch (NumberFormatException e) { return null; }
277                if (parts[i] < 0 || parts[i] > 255) return null;
278            }
279            for (int i = 0; i < 4; i++) address[i] = (byte) parts[i];
280            int port = parts[4] << 8 | parts[5];
281            InetAddress inetAddress;
282            try { inetAddress = InetAddress.getByAddress(address); }
283            catch (UnknownHostException e) { return null; }
284            return new InetSocketAddress(inetAddress, port);
285        }
286    
287        private static int choosePassivePort(int low, int high) {
288            int length = high - low;
289            int offset = (int) (currentTimeMillis() % length);
290            return low + offset;
291        }

293        private static void closeQuetly(Socket socket) {
294            try {
295                socket.close();
296            } catch (IOException e) {
297                // ignore
298            }
299        }
300    
301        private static void closeQuetly(ServerSocket socket) {
302            try {
303                socket.close();
304            } catch (IOException e) {
305                // ignore
306            }
307        }
